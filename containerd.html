# containerd

# 配置

谷歌 Kubernetes k8s.gcr.io 使用国内阿里源。

echo '
server = "https://k8s.gcr.io"

[host."https://registry.aliyuncs.com/v2/google_containers"]
  capabilities = ["pull", "resolve"]
  override_path = true
' | sudo tee /etc/containerd/certs.d/k8s.gcr.io/hosts.toml
